🔍 SQL Query Breakdown

SELECT 
    w.WorkerID, w.Name, w.Role,
    COUNT(DISTINCT wg.GearID) AS AssignedGearTypes,
    LISTAGG(sg.GearType, ', ') WITHIN GROUP (ORDER BY sg.GearType) AS GearAssigned,
    COUNT(CASE WHEN wg.Condition = 'Good' THEN wg.GearID END) AS GoodCondition,
    COUNT(CASE WHEN wg.Condition IN ('Fair', 'Poor', 'Damaged') THEN wg.GearID END) AS DamagedGear,
    ROUND(COUNT(CASE WHEN wg.Condition = 'Good' THEN 1 END) * 100 / 
          NULLIF(COUNT(DISTINCT wg.GearID), 0), 2) AS QualityPercent,
    CASE 
        WHEN COUNT(DISTINCT wg.GearID) = 0 THEN 'NO GEAR ASSIGNED'
        WHEN QualityPercent < 70 THEN 'REPLACE GEAR'
        ELSE 'COMPLIANT'
    END AS ComplianceStatus
FROM Worker w
LEFT JOIN Worker_Gear wg ON w.WorkerID = wg.WorkerID
LEFT JOIN Safety_Gear sg ON wg.GearID = sg.GearID
GROUP BY w.WorkerID, w.Name, w.Role;
